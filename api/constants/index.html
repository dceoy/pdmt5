def get_copy_ticks_name(value: int, *, prefer_alias: bool = False) -> str:
    """Return the MT5 COPY_TICKS name for an integer value.

    Args:

    """
    return _COPY_TICKS_FAMILY.name_by_value(value=value, prefer_alias=prefer_alias)
